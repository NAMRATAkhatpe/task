 8) Palindrome(int n)
 import java.util.Scanner;
 class Palindrome
{
 int n;
     Palindrome()
 {
        n=5;
 }
      Palindrome(int n)
 {
      this.n=n;
 }
      void display(){
    int temp=n,sum=0;
         while(n>0){
       int digit=n%10;
          sum=(sum*10)+digit;
           n/=10;
 }
 if(temp==sum)
         System.out.println("Number is Palindrome");
 else
           System.out.println("Number is not Palindrome");
 }
 }
         public class Main{
    public static void main(String[] args)
 {
        Palindrome p=new Palindrome();
